% Frames
\newmdenv[
  frametitlebackgroundcolor=LightGrey,%
  frametitlerule=true,%
  backgroundcolor=LighterGrey,%
  linecolor=LineLightGrey%
]{procedureFrame}

\newmdenv[
  frametitlebackgroundcolor=LightGrey,%
  frametitlerule=true,%
  backgroundcolor=LighterGrey,%
  linecolor=LineLightGrey%
]{exampleFrame}

% Environments
\newcounter{procedure}[section]
\newenvironment{procedure}[1][]{
  \refstepcounter{procedure}
  \begin{procedureFrame}[frametitle=\GetTranslation{procedure}~\theprocedure]
}{
  \end{procedureFrame}
}

\newcounter{example}[section]
\newenvironment{example}[1][]{
  \refstepcounter{example}
  \begin{exampleFrame}[frametitle=\GetTranslation{example}~\theprocedure]
}{
  \end{exampleFrame}
}
